#name ⇒ String

Returns the (full) name. This can be set to any string containing alphabetic characters, hypens and spaces, up to 200 characters. It cannot be blank, and any leading or trailing whitespace will be stripped.

def self.included(mod)
  mod.validates :name,
                presence: true,
                format: /\A[[:alpha:] -]+\z/,
                length: { maximum: 200 }

  # Make sure no trailing whitespace on names
  mod.before_save { self.name = name.strip }
end
